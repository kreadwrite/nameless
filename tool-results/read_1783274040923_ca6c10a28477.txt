   386→func settingsEditingItems(data: PeerInfoScreenData?, state: PeerInfoState, context: AccountContext, presentationData: PresentationData, interaction: PeerInfoInteraction, isMyProfile: Bool) -> [(AnyHashable, [PeerInfoScreenItem])] {
   387→    guard let data = data else {
   388→        return []
   389→    }
   390→    
   391→    enum Section: Int, CaseIterable {
   392→        case help
   393→        case bio
   394→        case birthday
   395→        case info
   396→        case account
   397→        case logout
   398→    }
   399→    
   400→    var items: [Section: [PeerInfoScreenItem]] = [:]
   401→    for section in Section.allCases {
   402→        items[section] = []
   403→    }
   404→    
   405→    let ItemNameHelp = 0
   406→    let ItemBio: AnyHashable = AnyHashable("bio_edit")
   407→    let ItemBioHelp = 2
   408→    let ItemPhoneNumber = 3

   413→    let ItemPeerColor = 8
   414→    let ItemBirthday = 9

   416→    let ItemBirthdayRemove = 11
   417→    let ItemBirthdayHelp = 12

   420→    let ItemPeerChatAutomationHelp = 15
   421→    
   422→    items[.help]!.append(PeerInfoScreenCommentItem(id: ItemNameHelp, text: presentationData.strings.EditProfile_NameAndPhotoOrVideoHelp))
   423→    
   424→    if let cachedData = data.cachedData as? CachedUserData {
   425→        let currentBio = state.updatingBio ?? (cachedData.about ?? "")
   426→        items[.bio]!.append(PeerInfoScreenMultilineInputItem(id: ItemBio, text: currentBio, placeholder: presentationData.strings.UserInfo_About_Placeholder, textUpdated: { updatedText in
   427→            interaction.updateBio(updatedText)

   429→            interaction.dismissInput()

   431→

   436→            case .enableEveryone:

   438→            case .enableContacts:
   439→                bioPrivacyInfo = presentationData.strings.Settings_About_PrivacyHelpContacts

   441→                bioPrivacyInfo = presentationData.strings.Settings_About_PrivacyHelpNobody

   445→            interaction.openBioPrivacy()

   448→    
   449→    var birthday: TelegramBirthday?
   450→    if let updatingBirthDate = state.updatingBirthDate {
   451→        birthday = updatingBirthDate
   452→    } else {
   453→        birthday = (data.cachedData as? CachedUserData)?.birthday
   454→    }
   455→    
   456→    var birthDateString: String
   457→    if let birthday {
   458→        birthDateString = stringForCompactBirthday(birthday, strings: presentationData.strings)
   459→    } else {
   460→        birthDateString = presentationData.strings.Settings_Birthday_Add
