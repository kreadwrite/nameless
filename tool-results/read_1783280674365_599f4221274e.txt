   102→private extension UserDefaults {
   103→    func namelessBool(_ key: String, default defaultValue: Bool = false) -> Bool {
   104→        if self.object(forKey: key) == nil {
   105→            return defaultValue
   106→        }
   107→        return self.bool(forKey: key)
   108→    }
   109→
   110→    func namelessInt32(_ key: String, default defaultValue: Int32 = 0) -> Int32 {
   111→        if self.object(forKey: key) == nil {
   112→            return defaultValue
   113→        }
   114→        return Int32(self.integer(forKey: key))
   115→    }
   116→
   117→    func namelessInt64(_ key: String, default defaultValue: Int64 = 0) -> Int64 {
   118→        if self.object(forKey: key) == nil {
   119→            return defaultValue
   120→        }
   121→        return Int64(self.integer(forKey: key))
   122→    }
   123→
   124→    func namelessDouble(_ key: String, default defaultValue: Double = 0.0) -> Double {
   125→        if self.object(forKey: key) == nil {
   126→            return defaultValue
   127→        }
   128→        return self.double(forKey: key)
   129→    }
   130→
   131→    func namelessString(_ key: String, default defaultValue: String = "") -> String {
   132→        return self.string(forKey: key) ?? defaultValue
   133→    }
   134→
   135→    func namelessStringArray(_ key: String) -> [String] {
   136→        return self.stringArray(forKey: key) ?? []
   137→    }
   138→}

   151→    func beginNamelessRollbackSnapshot() {
   152→        let values = storage.dictionaryRepresentation().filter { key, _ in
   153→            for prefix in NamelessRollbackStorage.managedPrefixes {
   154→                if key.hasPrefix(prefix) {
   155→                    return true
   156→                }
   157→            }
   158→            return false
   159→        }
   160→
   161→        if let data = try? JSONSerialization.data(withJSONObject: values, options: []) {
   162→            storage.set(data, forKey: NamelessRollbackStorage.snapshotKey)
   163→        }
   164→    }
   165→
   166→    @discardableResult
   167→    func restoreNamelessRollbackSnapshot() -> Bool {
   168→        guard let data = storage.data(forKey: NamelessRollbackStorage.snapshotKey),
   169→              let values = try? JSONSerialization.jsonObject(with: data, options: []) as? [String: Any] else {
   170→            return false
   171→        }
   172→
   173→        let existing = storage.dictionaryRepresentation()
   174→        for key in existing.keys {
   175→            for prefix in NamelessRollbackStorage.managedPrefixes {
   176→                if key.hasPrefix(prefix) {
   177→                    storage.removeObject(forKey: key)
   178→                    break
   179→                }
   180→            }
   181→        }
   182→
   183→        for (key, value) in values {
   184→            storage.set(value, forKey: key)
   185→        }
   186→
   187→        storage.synchronize()
   188→        return true
   189→    }
